Template Stack on C++ 
template <class Type>
class Stack
{
private:

	Type st[MAX];
	int top;
public:

	Stack();
	void push(Type var);
	Type pop();
};

template<class Type>
Stack<Type>::Stack()
{
	top = -1;
}

template<class Type>
void Stack<Type>::push(Type var)
{
	st[++top] = var;
}

template<class Type>
Type Stack<Type>::pop()
{
	return st[top--];
}
